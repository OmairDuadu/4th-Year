--Company table 
create table Companies(
  Internal_ID varchar (100) Primary key Unique not null,
  Company_Name varchar (100) Unique not null,
  Annual_Revenue money not null
  );

create table Traders(
  Staff_ID varchar (100) Primary key Unique not null,
  Trader_Name varchar (100) not null,
  join_Date date not null
  );

create table Portfolios(
  Portfolio_ID varchar (100) Primary key Unique not null,
  Portfolio_Name varchar (100) not null
  );

create table Prices(
  Internal_ID varchar (100) Primary key Unique not null,
  Date_Price date not null,
  Stock_Value money not null,
  Foreign key (Internal_ID)
   references Companies (Internal_ID)
);

--Positions Table
create table Positions(
  Internal_ID varchar (100) not null,
  Portfolio_ID varchar (100) not null,
  Current_Stock int not null,
  Submitted_date timestamp not null,
  primary key (Internal_ID, Portfolio_ID),
  Foreign key (Portfolio_ID)
   references Portfolios (Portfolio_ID),
  Foreign key (Internal_ID)
   references Companies (Internal_ID)
  );
